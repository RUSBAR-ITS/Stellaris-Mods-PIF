# PIF building category: LIFECYCLE
# Vanilla building: holding_park_ranger_lodge
# Vanilla class: holding
# Vanilla source: buildings/15_overlord_holdings.txt

destroy_trigger = {
	OR = {
		is_planet_class = pc_city
		is_planet_class = pc_hive
		is_planet_class = pc_machine
		is_planet_class = pc_relic
	}
}

on_built = {
	planet = {
		add_blocker = {
			type = d_nature_preserve_blocker
		}
	}
}

on_destroy = {
	planet = {
		if = {
			limit = {
				has_deposit = d_nature_preserve_blocker
			}
			remove_deposit = d_nature_preserve_blocker
		}
	}
}
