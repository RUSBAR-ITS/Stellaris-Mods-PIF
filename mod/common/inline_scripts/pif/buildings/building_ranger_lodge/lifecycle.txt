# PIF building category: LIFECYCLE
# Vanilla building: building_ranger_lodge
# Vanilla class: regular
# Vanilla source: buildings/05_research_buildings.txt

destroy_trigger = {
	exists = owner
	OR = {
		owner = {
			is_regular_empire = no
		}
		has_modifier = slave_colony
		owner = {
			NOT = {
				has_valid_civic = civic_environmentalist
			}
		}
		is_planet_class = pc_city
		is_planet_class = pc_hive
		is_planet_class = pc_machine
		is_planet_class = pc_relic
	}
}

on_built = {
	planet = {
		add_blocker = {
			type = d_big_nature_preserve_blocker
		}
	}
}

on_destroy = {
	planet = {
		if = {
			limit = {
				has_deposit = d_big_nature_preserve_blocker
			}
			every_deposit = {
				limit = {
					is_deposit_type = d_big_nature_preserve_blocker
				}
				remove_deposit = d_big_nature_preserve_blocker
			}
		}
	}
}
