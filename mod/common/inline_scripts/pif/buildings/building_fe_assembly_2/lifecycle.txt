# PIF building category: LIFECYCLE
# Vanilla building: building_fe_assembly_2
# Vanilla class: regular
# Vanilla source: buildings/13_fallen_empire_buildings.txt

destroy_trigger = {
	exists = owner
	owner = {
		is_fallen_empire = no
		is_machine_empire = no
		OR = {
			is_regular_empire = no
			has_policy_flag = robots_outlawed
		}
	}
}

on_queued = {
	if = {
		limit = {
			is_variable_set = local_spent_biomass
			any_owned_pop_group = {
				has_trait = trait_wilderness
				pop_group_size > local_spent_biomass
			}
		}
		random_owned_pop_group = {
			limit = {
				has_trait = trait_wilderness
				pop_group_size > local_spent_biomass
			}
			kill_pop_group = {
				pop_group = this
				amount = local_spent_biomass
			}
		}
	}
	else_if = {
		limit = {
			is_variable_set = local_spent_biomass
		}
		owner = {
			global_biomass_spender = yes
		}
	}
	owner = {
		subtract_variable = {
			which = fe_assembly_1_upgrade_count
			value = 1
		}
	}
}

on_unqueued = {
	unqueue_biomass_effect = yes
	owner = {
		change_variable = {
			which = fe_assembly_1_upgrade_count
			value = 1
		}
	}
}
